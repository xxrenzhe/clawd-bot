## Helm Installation

### Add Repository

```bash
helm repo add moltbot https://charts.molt.bot
helm repo update
```

### Install

```bash
helm install moltbot moltbot/moltbot \
  --namespace moltbot \
  --create-namespace \
  --set secrets.anthropicApiKey=$ANTHROPIC_API_KEY \
  --set secrets.telegramToken=$TELEGRAM_BOT_TOKEN
```

### Values File

```yaml
# values.yaml
replicaCount: 3

image:
  repository: moltbot/moltbot
  tag: latest
  pullPolicy: IfNotPresent

resources:
  requests:
    memory: "512Mi"
    cpu: "500m"
  limits:
    memory: "1Gi"
    cpu: "1000m"

autoscaling:
  enabled: true
  minReplicas: 2
  maxReplicas: 10
  targetCPUUtilizationPercentage: 70

secrets:
  anthropicApiKey: ""
  telegramToken: ""
  gatewayPassword: ""

config:
  gateway:
    bind: "0.0.0.0"
    port: 18789
  agent:
    provider: anthropic
    model: claude-sonnet-4-20250514

redis:
  enabled: true
  architecture: standalone

persistence:
  enabled: true
  size: 10Gi
  storageClass: ""
```

### Install with Values

```bash
helm install moltbot moltbot/moltbot \
  --namespace moltbot \
  --create-namespace \
  -f values.yaml
```

## Raw Manifests

### Namespace

```yaml
apiVersion: v1
kind: Namespace
metadata:
  name: moltbot
  labels:
    app: moltbot
```

### Secret

```yaml
apiVersion: v1
kind: Secret
metadata:
  name: moltbot-secrets
  namespace: moltbot
type: Opaque
stringData:
  ANTHROPIC_API_KEY: "sk-ant-..."
  TELEGRAM_BOT_TOKEN: "123456:ABC..."
  GATEWAY_PASSWORD: "strong-password"
```

### ConfigMap

```yaml

        "model": "claude-sonnet-4-20250514"
```

### Deployment

```yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: moltbot
  namespace: moltbot
spec:
  replicas: 3
  selector:
    matchLabels:
      app: moltbot
  template:
    metadata:
      labels:
        app: moltbot
    spec:
      containers:
      - name: moltbot
        image: moltbot/moltbot:latest
        ports:
        - containerPort: 18789
        envFrom:
        - secretRef:
            name: moltbot-secrets
        volumeMounts:
        - name: config
          mountPath: /app/config
        - name: data
          mountPath: /app/data
        resources:
          requests:
            memory: "512Mi"
            cpu: "500m"
          limits:
            memory: "1Gi"
            cpu: "1000m"
        livenessProbe:
          httpGet:
            path: /health
            port: 18789
          initialDelaySeconds: 30
          periodSeconds: 10
        readinessProbe:
          httpGet:
            path: /health
            port: 18789
          initialDelaySeconds: 5
          periodSeconds: 5
      volumes:
      - name: config
        configMap:
          name: moltbot-config
      - name: data
        persistentVolumeClaim:
          claimName: moltbot-data
```

## Horizontal Pod Autoscaler

```yaml
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: moltbot
  namespace: moltbot
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: moltbot
  minReplicas: 2
  maxReplicas: 10
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 70
  - type: Resource
    resource:
      name: memory
      target:
        type: Utilization
        averageUtilization: 80
```

## Operations

### View Status

```bash
# Pod status
kubectl get pods -n moltbot

# Logs
kubectl logs -n moltbot -l app=moltbot -f

# Resource usage
kubectl top pods -n moltbot
```

### Rolling Update

```bash
# Update image
kubectl set image deployment/moltbot \
  moltbot=moltbot/moltbot:2.4.0 \
  -n moltbot

# Check rollout status
kubectl rollout status deployment/moltbot -n moltbot

# Rollback if needed
kubectl rollout undo deployment/moltbot -n moltbot
```

## Troubleshooting

| Issue | Cause | Solution |
|-------|-------|----------|
| Pods pending | Resource limits | Check node capacity |
| CrashLoopBackOff | Config error | Check logs, secrets |
| Connection refused | Service issue | Verify endpoints |
| High latency | Overloaded | Scale up replicas |

### Debug Commands

```bash
# Describe pod
kubectl describe pod -n moltbot <pod-name>

# Shell into pod
kubectl exec -it -n moltbot <pod-name> -- sh

# Check events
kubectl get events -n moltbot --sort-by='.lastTimestamp'
```
